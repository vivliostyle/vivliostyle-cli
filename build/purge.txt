# Install-time-only packages, force-purged after mmdebstrap assembles the rootfs
# (pass 1; see purge-late.txt for pass 2). build/audit.ts orders and applies this
# list; it does not decide its contents.
#
# These are NOT orphans an automatic tool would drop: the apt/dpkg + browser
# dependency graph keeps almost everything reachable, so each entry below is a
# deliberate "reachable but unwanted at runtime" decision, with its reason. The
# list is curated and re-verified out of band by build/audit-removable.ts -- run
# it on a Debian bump / dependency change and reconcile the diff (that script's
# header documents exactly how). image-contract.sh is the proof the image still
# works without these.

# update-rc.d / invoke-rc.d helpers; the image runs no services.
init-system-helpers

# system software-GL -- Chrome uses its own bundled SwiftShader instead.
mesa-libgallium
libllvm19
libz3-4

# GTK icon set -- surfaces only in GTK dialogs.
adwaita-icon-theme

# NodeSource pulls python in so node-gyp is available, but ships no compiler;
# building native addons at "npm install" needs extra packages anyway, so
# python alone is pointless.
# see https://github.com/nodejs/node-gyp/blob/v12.3.0/README.md?plain=1#L72
python3
python3-minimal
python3.13
python3.13-minimal

# gnupg. apt verifies repository signatures with gpgv (a separate, kept package);
# full gnupg / key management is never used. libassuan9 stays via
# poppler-utils -> libpoppler -> libgpgmepp6t64 -> libgpgme11t64.
gnupg
gnupg-l10n
gpg
gpg-agent
gpgsm
gpgconf
dirmngr
pinentry-curses
libnpth0t64
libksba8

# reachable via Depends but unused by the render path: the GTK icon-cache tool,
# X bitmap-font utils, the dconf settings backend, lm-sensors and the AppArmor
# userspace library.
gtk-update-icon-cache
libfontenc1
xfonts-utils
xfonts-encodings
dconf-service
dconf-gsettings-backend
libdconf1
libsensors5
libsensors-config
libapparmor1

# PAM, account management and init: nothing runs as a service or authenticates in
# the image, and useradd/groupadd were only needed at build time (the vivliostyle
# user already exists).
libpam-systemd
libpam-modules
libpam-modules-bin
libpam-runtime
libpam0g
passwd
adduser
login.defs
systemd
systemd-sysv
sysvinit-utils
